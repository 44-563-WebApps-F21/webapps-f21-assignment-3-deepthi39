In computer science, graph traversal (also known as graph search) refers to the process of visiting (checking and/or updating) each vertex in a graph. Such traversals are classified by the order in which the vertices are visited. Tree traversal is a special case of graph traversal. link to source code 
            vector> adj;  // adjacency list representation
                int n; // number of nodes
                int s; // source vertex
                
                queue q;
                vector used(n);
                vector d(n), p(n);
                
                q.push(s);
                used[s] = true;
                p[s] = -1;
                while (!q.empty()) {
                    int v = q.front();
                    q.pop();
                    for (int u : adj[v]) {
                        if (!used[u]) {
                            used[u] = true;
                            q.push(u);
                            d[u] = d[v] + 1;
                            p[u] = v;
                        }
                    }
